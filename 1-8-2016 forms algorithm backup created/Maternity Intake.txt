function ceil(x) {return x > parseInt(x) ? parseInt(x) + 1 : parseInt(x);}

function UltraSoundField (nlist) {
	var self = this;
	self.n = nlist;
	self.dateN = self.n+"-list.value_1";
	self.weekN = self.n+"-list.value_3";
	self.deliveryN = self.n+"-list.value_4";
	self.hiddenN = self.n+"-list.value_5";

	self.getDate = function () {return getValue(self.dateN);};
	self.getNWeeks = function() {return getValue(self.weekN);};
	self.setNWeeks = function (wks) {setValue(self.weekN, wks);};

	self.clear = function() {
		if (getValue(self.dateN) != "")
			setValue(self.dateN, "");

		if (getValue(self.weekN) != "")
			setValue(self.weekN, "");

		if (getValue(self.deliveryN) != "")
			setValue(self.deliveryN, "");

		if(getValue(self.hiddenN) != "")
			setValue(self.hiddenN, "");
	};

	self.onUsChange = function() {
		var usDate = new Date(self.getDate());
		var nWeeks = self.getNWeeks();
		var delivery;

		if (usDate == null || usDate == "Invalid Date") {
			self.clear();
			return;
		}

		if (usDate > new Date()) {
			alert("Ultraound date cannot be in future.");
			self.clear();
			return;
		}

		if (nWeeks == null || nWeeks == "") {
			setValue(self.deliveryN, "");
			setValue(self.hiddenN, "");
			return;
		}

		setValue(self.hiddenN, 159618);
		delivery = new Date(usDate.getTime() +(40-nWeeks)*(1000*3600*24*7));

		delivery.setUTCHours(0);
		delivery.setUTCMinutes(0);
		delivery.setUTCSeconds(0);
		delivery.setUTCMilliseconds(0);

		setValue(self.deliveryN, parseDateFromUTCJsToString("yyyy-mm-dd", delivery));
	}


	getField(self.dateN).change(function(){self.onUsChange();});
	getField(self.weekN).change(function(){self.onUsChange();});
	setValue(self.hiddenN, "");
}

function onLMPChange () {
	var lastLMP = new Date(getValue("OB_LMP.value"));
	var nweeks = 0;
	var daliveryDate;
	var now = new Date();

	if (lastLMP == null || lastLMP == "Invalid Date") {
		setValue("OB_EDD.value", "");
		setValue("OB_gest.value", "");
		setValue("OB_gestcalc.value", "");

		setValue("OB_LMP.value", "");

                //getField("OB_LMP.error").show().focus();
		//onLMPChange();
                alert("LMP cannot be in future.");
                //getField("OB_LMP.value").focus();
                //getField("OB_LMP.value").triggerHandler( "focus" );
		return;
	}

        //setValue("OB_LMP.error", "");
        //getField("OB_LMP.error").hide();

	now.setUTCHours(0);
	now.setUTCMinutes(0);
	now.setUTCSeconds(0);
	now.setUTCMilliseconds(0);

function parseDateFromUTCJsToString(sFormat, jsDate) {
 	if(jsDate == null) {
 		jsDate = new Date();
 	}
 	return sFormat.replace(/mm/, tenrule(jsDate.getUTCMonth()+1)).replace(/dd/, tenrule(jsDate.getUTCDate())).replace(/yyyy/, (jsDate.getUTCFullYear() ? jsDate.getUTCFullYear() : 1900+jsDate.getUTCYear()));
}

function isBlank(listValueString,numListElements,allBlank) {
    var numCheck = 1;
    if (allBlank) {numCheck = numListElements};
    var thisListValueString = listValueString;
    if (numListElements > 1) {
        thisListValueString = thisListValueString + "_1";
    };
    var i = 1;
    while (numCheck >= i) {
        if (getValue(thisListValueString) != '' ) {
            return false;
        };
        i++;
        thisListValueString = listValueString + "_" + i;
    };
    return true;
};

function removeBlanks(startRow,numberRows,listValueString,numListElements,allBlank) {
    //
    //function shuffles all non-blank rows from array item start to start+number-1 to the beginning
    // startRow is the first row in the array
    // numberRows is the number of rows in this block
    // numListElements is the number of elements in each row
    // if numListElements is more than one allBlank TRUE means to ensure all elements are blank otherwise checks just the first
    //

    var i = startRow;
    var nonBlankCount = 0;
    var lastRow = startRow +numberRows - 1;
    var baseListValueString = listValueString;
    if (numListElements > 1) {
        baseListValueString = baseListValueString + "_1";
    }

    while (lastRow >= i) {

        // keep going if nonblank
        if (!isBlank((i+listValueString),numListElements,allBlank)) {
            i++;
            nonBlankCount ++;
        } else {
            // found blank line, find next non-blank
            var j = i + 1;
            while ((lastRow >= j) &amp;&amp; (isBlank((j+listValueString),numListElements,allBlank))) {
                j++;
            }

            // if we have found a nonblank, move the nonblanks down
            var newStart = j;

            while (!isBlank((j+listValueString),numListElements,allBlank) &amp;&amp; (lastRow >= j)) {
                //move all of the elements in this row
                    for (k=1; numListElements >=k; k++) {
                        var thisListElement = listValueString + "_" + k;
                        if ( (k==1) &amp;&amp; (numListElements == 1) ) {
                            thisListElement = listValueString;
                        };
                    var moveValuestring = getValue(j + thisListElement);
                    setValue(i + thisListElement, moveValuestring);
                    setValue(j + thisListElement,'');
                };

                nonBlankCount++;
                i++;
                j++;
            };

            if (j > lastRow) {
                i = j;
            };
        };
    };
    console.log(startRow,nonBlankCount);
    return nonBlankCount;
};   //end function

	    addedCount = 1;
